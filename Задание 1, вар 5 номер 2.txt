#include <iostream>
#include <cmath>
using namespace std;
void random(int(*matrix)[15], int n, int m)
{
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			matrix[i][j] = rand() % 100 - 80;
			cout << matrix[i][j] << "\t";
		}
		cout << "\n";
	}
}
void TANsumma(int(*matrix)[15], int n, int m)
{
	double tg;
	int sum = 0;
	for (int i = 0; i < n; i++)
	{

		for (int j = 0; j < m; j++) {
			if (matrix[i][j] > 0) {
				sum = sum + matrix[i][j];
			}
			
		}

		
	}
	cout << "s=" << sum << " ";
	tg = sin(sum) / cos(sum);
	cout << "tan=" << tg << " ";

}
